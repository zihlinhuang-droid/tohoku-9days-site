/* --- 3-1) 行程卡片 --- */

function renderTripCard(item) {

  const typeKey = item.type || "fixed";

  const badge = `<span class="badge ${TYPE_CLASS[typeKey] || TYPE_CLASS.fixed}">${TYPE_LABEL[typeKey] || "固定"}</span>`;

  const timeEl = item.time ? `<span class="trip-card__time">${escapeHTML(item.time)}</span>` : "";

  const foodsJson = JSON.stringify(item.foods || []);


  // 按鈕：Google Maps + 附近吃食

  const mapBtn = item.mapLink

    ? `<a class="btn-chip btn-chip--map" href="${escapeHTML(item.mapLink)}" target="_blank" rel="noopener">📍 Google Maps</a>`

    : "";

  const foodBtn = (item.foods && item.foods.length)

    ? `<button class="btn-chip btn-chip--food"

         data-food='${escapeHTML(foodsJson)}'

         data-food-title="${escapeHTML(item.place)} · 附近吃食">

         🍜 附近吃食

       </button>`

    : "";


  return `

    <article class="trip-card">

      <div class="trip-card__head">

        ${badge}${timeEl}

        <h4 class="trip-card__name">${escapeHTML(item.place)}</h4>

      </div>

      ${item.description ? `<p class="trip-card__desc">${escapeHTML(item.description)}</p>` : ""}

      ${item.note ? `<p class="trip-card__note">${escapeHTML(item.note)}</p>` : ""}

      ${(mapBtn || foodBtn) ? `<div class="trip-card__actions">${mapBtn}${foodBtn}</div>` : ""}

    </article>

  `;

}

/* --- 4) Day 導覽互動 --- */

function bindDayNav() {

  const navEl = document.getElementById("dayNav");

  const navInner = document.getElementById("dayNavInner");

  const btns = document.querySelectorAll(".day-nav__btn");


  btns.forEach(btn => {

    btn.addEventListener("click", () => {

      const target = document.getElementById(btn.dataset.target);

      if (!target) return;

      const y = target.getBoundingClientRect().top + window.pageYOffset - navEl.offsetHeight + 1;

      window.scrollTo({ top: y, behavior: "smooth" });

    });

  });


  const dayIds = Array.from(btns).map(b => b.dataset.target);

  function updateActive() {

    const scrollY = window.scrollY + navEl.offsetHeight + 20;

    let current = dayIds[0];

    dayIds.forEach(id => {

      const el = document.getElementById(id);

      if (el && el.offsetTop <= scrollY) current = id;

    });

    btns.forEach(b => {

      const isActive = b.dataset.target === current;

      b.classList.toggle("is-active", isActive);

      // 讓當前按鈕自動捲入可視範圍（手機橫向 tab 的體驗提升）

      if (isActive) {

        const btnLeft = b.offsetLeft;

        const btnRight = btnLeft + b.offsetWidth;

        const viewLeft = navInner.scrollLeft;

        const viewRight = viewLeft + navInner.clientWidth;

        if (btnLeft < viewLeft + 12) navInner.scrollTo({ left: btnLeft - 12, behavior: "smooth" });

        else if (btnRight > viewRight - 12) navInner.scrollTo({ left: btnRight - navInner.clientWidth + 12, behavior: "smooth" });

      }

    });

  }

  window.addEventListener("scroll", updateActive, { passive: true });

  updateActive();

}


/* --- 5) Modal（附近吃食）--- */

function bindModal() {

  const modal = document.getElementById("foodModal");

  const titleEl = document.getElementById("foodModalTitle");

  const listEl = document.getElementById("foodModalList");


  document.querySelectorAll(".btn-chip--food").forEach(btn => {

    btn.addEventListener("click", () => {

      const title = btn.dataset.foodTitle || "附近吃食";

      let items = [];

      try { items = JSON.parse(btn.dataset.food || "[]"); }

      catch (e) { console.error("food JSON parse error:", e); }


      titleEl.textContent = title;

      listEl.innerHTML = items.length

        ? items.map(item => `

            <li class="modal__item">

              <div class="modal__item-head">

                <span class="modal__item-name">${escapeHTML(item.name || "")}</span>

                ${item.mapLink ? `<a class="modal__item-map" href="${escapeHTML(item.mapLink)}" target="_blank" rel="noopener">Google Maps</a>` : ""}

              </div>

              ${item.info ? `<div class="modal__item-desc">${escapeHTML(item.info)}</div>` : ""}

            </li>

          `).join("")

        : `<li class="modal__item"><div class="modal__item-desc" style="color:var(--color-text-light)">尚無資料，可於 itineraryData 對應景點的 foods 陣列新增。</div></li>`;


      modal.classList.add("is-open");

      document.body.style.overflow = "hidden";

    });

  });


  modal.querySelectorAll("[data-modal-close]").forEach(el => {

    el.addEventListener("click", closeModal);

  });

  document.addEventListener("keydown", e => {

    if (e.key === "Escape" && modal.classList.contains("is-open")) closeModal();

  });

  function closeModal() {

    modal.classList.remove("is-open");

    document.body.style.overflow = "";

  }

}
